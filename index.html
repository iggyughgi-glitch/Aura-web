/** * AURA SEARCH PROXY * Vercel Serverless Function — /api/search * * Architecture: * Frontend → /api/search?q=... (HTTPS, same origin, no CORS issue) * This function → SearXNG instance (server-side, IP never exposed) * SearXNG → 70+ engines simultaneously, ads/trackers stripped * Response → cleaned JSON, minimum surface area, back to frontend * * The SearXNG instance URL lives ONLY in Vercel environment variables. * It never touches the client. Zero exposure. */ module.exports = async function handler(req, res) { // ── Security headers ────────────────────────────────────────────── res.setHeader('X-Content-Type-Options', 'nosniff'); res.setHeader('X-Frame-Options', 'DENY'); res.setHeader('X-XSS-Protection', '1; mode=block'); res.setHeader('Referrer-Policy', 'no-referrer'); res.setHeader('Cache-Control', 'no-store, no-cache, must-revalidate'); res.setHeader('Permissions-Policy', 'geolocation=(), camera=(), microphone=()'); // CORS: only allow requests from your own domain const origin = req.headers.origin || ''; if (origin.includes('vercel.app') || origin.includes(process.env.ALLOWED_ORIGIN || '')) { res.setHeader('Access-Control-Allow-Origin', origin); } // ── Method guard ────────────────────────────────────────────────── if (req.method !== 'GET') { return res.status(405).json({ error: 'Method not allowed' }); } // ── Input sanitization ──────────────────────────────────────────── const raw = req.query.q; if (!raw || typeof raw !== 'string' || raw.trim().length === 0) { return res.status(400).json({ error: 'Query required' }); } // Strip anything dangerous, cap length const query = raw.trim().replace(/[<>"'`]/g, '').slice(0, 200); if (query.length === 0) { return res.status(400).json({ error: 'Invalid query' }); } // ── SearXNG instance — lives in Vercel env var, never in code ───── const SEARXNG = process.env.SEARXNG_INSTANCE; if (!SEARXNG) { console.error('SEARXNG_INSTANCE env var not set'); return res.status(503).json({ error: 'Search engine offline' }); } // ── Build SearXNG request ───────────────────────────────────────── // JSON format, English, safe search on, web category const params = new URLSearchParams({ q: query, format: 'json', language: 'en', safesearch: '1', categories: 'general', }); const searchURL = `${SEARXNG}/search?${params.toString()}`; try { const upstream = await fetch(searchURL, { method: 'GET', headers: { // Headless browser identity — no referrer, no tracking 'User-Agent': 'Mozilla/5.0 (compatible; AuraBot/1.0; +https://aura.app)', 'Accept': 'application/json', 'Accept-Language': 'en-US,en;q=0.9', // Never send referrer upstream 'Referer': '', }, // Hard timeout — 8 seconds signal: AbortSignal.timeout(8000), }); if (!upstream.ok) { throw new Error(`SearXNG responded ${upstream.status}`); } const data = await upstream.json(); // ── Data ingestion: strip to minimum surface area ───────────── // Only pass through what the UI needs. No tracking data, no // engine metadata, no raw scores — nothing the client doesn't use. const results = (data.results || []) .filter(r => r.url && r.title) // must have both .slice(0, 8) // max 8 results .map((r, i) => ({ id: String(i + 1).padStart(2, '0'), title: sanitizeText(r.title || 'Result'), desc: sanitizeText(r.content || r.title || ''), url: sanitizeURL(r.url), // domain shown in card footer domain: extractDomain(r.url), })); return res.status(200).json({ results }); } catch (err) { // Never expose internal error details to client console.error('Search proxy error:', err.message); return res.status(500).json({ error: 'Search temporarily unavailable' }); } } // ── Helpers ─────────────────────────────────────────────────────────── function sanitizeText(str) { return String(str) .replace(/<[^>]*>/g, '') // strip any HTML tags .replace(/&/g, '&') .replace(/</g, '<') .replace(/>/g, '>') .replace(/"/g, '"') .trim() .slice(0, 200); } function sanitizeURL(str) { try { const u = new URL(str); // Only allow http/https — no javascript:, data:, etc. if (u.protocol !== 'https:' && u.protocol !== 'http:') return '#'; return u.href; } catch { return '#'; } } function extractDomain(str) { try { return new URL(str).hostname.replace('www.', ''); } catch { return ''; } }
